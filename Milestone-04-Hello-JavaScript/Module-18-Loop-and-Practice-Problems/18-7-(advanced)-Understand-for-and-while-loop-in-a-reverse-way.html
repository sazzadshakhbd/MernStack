For Loop - Reverse Way

  1. create reverse.js file
  2. Display 1-10
  3. var number = 0;
while (number <= 10) {
    console.log(number);
    number++;
}
  4. Display 10 - 1
  5. var number = 10;
while (number >= 1) {
    console.log(number);
    number--;
}

  6. for (var i = 10; i >= 1; i--) {
    console.log(i);
}
